create table test.person (
    id serial primary key ,
    linkPhoto text not null unique,
	timeInput timestamp,
	timeOut timestamp
);

create table test.camera (
	id serial primary key,
	model text not null,
	x_central real not null,
	y_central real not null
);

create table test.photo (
	id serial primary key,
	idCamera integer not null references test.camera(id),
	idPerson integer not null references test.person(id),
	linkPhoto text not null unique,
	timePhoto timestamp
);


create table test.blindLine(
	x1 real not null,
	y1 real not null,
	x2 real not null,
	y2 real not null,
	idCamera1 integer not null references test.camera(id),
	idCamera2 integer not null references test.camera(id)
);
